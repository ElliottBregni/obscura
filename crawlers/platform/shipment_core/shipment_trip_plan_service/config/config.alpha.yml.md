flowchart TD
    Start([Start]) --> Walk[/"Walk repository (os.walk)"/]
    Walk --> Prune[/"Prune skip dirs & files"/]
    Prune --> Discover[/"Collect FileEntry objects"/]
    Discover --> ForEach{File needs processing?}
    ForEach -- Yes --> Read["Read source file contents"]
    Read --> Copilot["Call copilot -p to generate Mermaid"]
    Copilot --> Split["Split into diagrams (split_mermaid_diagrams)"]
    Split --> RenderChoice{Render via mmdc?}
    RenderChoice -- mmdc OK --> MMDC["Render with mmdc -> SVG"]
    RenderChoice -- mmdc fail --> Kroki["POST to Kroki -> SVG"]
    MMDC --> Embed["Embed Mermaid + inline SVG into .md"]
    Kroki --> Embed
    Embed --> Write["Write Markdown file (if not existing)"]
    Write --> Counter["Increment written counter"]
    Counter --> LoopBack[/Process next file/]
    ForEach -- No --> IndexGen["Generate INDEX.md"]
    IndexGen --> End([Done])
