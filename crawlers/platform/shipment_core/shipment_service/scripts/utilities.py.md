sequenceDiagram
    participant Caller
    participant get_session as get_session_and_engine_by_stage
    participant yaml

    participant get_carrier as get_carrier

    participant Organizations
    participant get_range as get_shipment_time_range

    Caller->>get_session: call(stage)
    get_session->>yaml: open & load ../lambdas/secrets.${stage}.yml
    yaml-->>get_session: config dict (FV_DB_HOST,...)
    get_session->>os: set FV_DB_HOST, FV_DB_NAME, FV_DB_USER, FV_DB_PASS
    get_session->>database: get_database_session_and_engine()
    database-->>get_session: (session, engine)
    get_session-->>Caller: return session, engine

    Caller->>get_shipments: call(session)
    get_shipments->>DB: now = utcnow()
    get_shipments->>ShipmentStops: query stop_sequence==1 and earliest_arrival_datetime < now+30min
    ShipmentStops-->>get_shipments: subscribe_shipments
    get_shipments->>ShipmentStops: query stop_sequence!=1 and latest_arrival_datetime > now
    ShipmentStops-->>get_shipments: future_shipments
    get_shipments->>get_shipments: intersect subscribe_shipments & future_shipments
    get_shipments-->>Caller: return set(expected_shipment_ids)

    Caller->>get_carrier: call(session, shipment_id)
    get_carrier->>OrganizationProfiles: query where code=="CR"
    OrganizationProfiles-->>get_carrier: carrier_profile id
    get_carrier->>ShipmentOrganizations: query where shipment_id==shipment_id
    ShipmentOrganizations-->>get_carrier: associated_orgs
    get_carrier->>Organizations: query where organization_profile_id==carrier_profile and id in associated_ids
    Organizations-->>get_carrier: carrier
    get_carrier-->>Caller: return carrier

    Caller->>get_range: call(session, shipment_id)
    get_range->>ShipmentStops: query where shipment_id==shipment_id
    ShipmentStops-->>get_range: stops list
    get_range->>get_range: compute min earliest_arrival_datetime and max latest_arrival_datetime
    get_range-->>Caller: return (start, stop)
