participant SQS as SQS_Event
participant Lambda as Lambda
participant DBConn as DB_CONN_ENTITY
participant Handler as BehindScheduleExceptionHandler

participant WorkerClear as clear_exception_worker

participant Logger as Logger

SQS ->> Lambda: Event with Records
Lambda ->> DBConn: establish_connection()
loop records
    Lambda ->> Handler: instantiate(DBConn, solution_id, entity_external_id)
    Handler ->> DB: is_behind_schedule_enabled(cursor, solution_id)
    alt not enabled
        Handler ->> Logger: info solution does not support behind schedule
    else enabled
        Handler ->> DB: get_org_id_from_solution_id(solution_id)
        alt org_id missing
            Handler ->> Logger: error "Org ID not found" and raise
            Lambda ->> Logger: error handling record
        else org_id present
            Handler ->> DB: get_behind_scheduled_data(cursor, entity_id, solution_id)
            alt entity exists
                Handler ->> DB: get_current_behind_schedule_exception(cursor, entity_id, solution_id)
                alt behind_schedule and no exception
                    Handler ->> Logger: info "set behind schedule exception"
                    Handler ->> WorkerSet: set_exception_worker(entity, solution_id, org_id)
                    WorkerSet ->> Invoker: invoke_set_exception(event, ..., fv_event, invoke_type)
                    Invoker -->> WorkerSet: response
                    alt response status >= 400
                        WorkerSet ->> Logger: error and raise Exception
                    else success
                        WorkerSet ->> Logger: info "Set behind schedule exception."
                    end
                else not behind_schedule and exception exists
                    Handler ->> Logger: info "clear behind schedule exception"
                    Handler ->> WorkerClear: clear_exception_worker(entity, solution_id, org_id, exception)
                    WorkerClear ->> Invoker: invoke_set_exception(event, ..., fv_event, exception_to_clear, invoke_type)
                    Invoker -->> WorkerClear: response
                    alt response status >= 400
                        WorkerClear ->> Logger: error and raise Exception
                    else success
                        WorkerClear ->> Logger: info "Cleared behind schedule exceptions."
                    end
                else no action
                    Handler ->> Logger: info "No exception to be set/cleared"
                end
            end
        end
    end
end
