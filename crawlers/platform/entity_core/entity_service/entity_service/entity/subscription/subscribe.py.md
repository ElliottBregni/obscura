```mermaid
sequenceDiagram
participant Client
participant Lambda as LambdaHandler
participant DB as DB_CONN
participant Auth as AuthModule
participant FVaws as FVawsLambdas
participant EntitySvc as EntityServiceDB
participant Invokinator as InvokinatorIAM
participant Subscribe as SubscribeInvoker

Client->>LambdaHandler: invoke lambda_handler(event)
LambdaHandler->>DB: establish_connection()
LambdaHandler->>DB: acquire cursor
LambdaHandler->>FVaws: get_event_body(event)
LambdaHandler->>FVaws: get_path_parameter(s), get_query_parameter(s), get_features(event)
LambdaHandler->>Auth: get_user_org_types(event)
alt solution_id present
    LambdaHandler->>Auth: get_path_parameter(entity_id)
    LambdaHandler->>LambdaHandler: set subscribing_product = FINISHED_VEHICLE
else dealer/internal path
    LambdaHandler->>EntitySvc: entity_dealer_authorization(event, cursor, [internal_entity_id], dealer_org_id)
    EntitySvc-->>LambdaHandler: entity_id, solution_id, _, organization_type
    LambdaHandler->>EntitySvc: get_system_configuration_setting(ALERT_VIN_ACTION, solution_id)
    alt alert action == hide
        LambdaHandler-->>Client: raise ForbiddenError("'Alert me' VIN action is not enabled")
    end
else carrier/partner path
    LambdaHandler->>EntitySvc: entity_carrier_authorization(event, cursor, internal_entity_id)
    EntitySvc-->>LambdaHandler: entity_id, solution_id
    LambdaHandler->>LambdaHandler: set subscribing_product = CARRIER_VIEW or INVENTORY_VIEW
end
LambdaHandler->>Invokinator: invoke_get_solution(solution_id)
Invokinator-->>LambdaHandler: solution (contains customer_id)
LambdaHandler->>EntitySvc: get_entity_description(cursor, entity_id, solution_id)
EntitySvc-->>LambdaHandler: description
LambdaHandler->>LambdaHandler: populate json_data with owner_id, reference_id, reference_description, subscribing_product, org_id
LambdaHandler->>Subscribe: invoke_subscribe_to_entity(event, json_data)
Subscribe->>FVaws: deepcopy event, set body and httpMethod='POST'
Subscribe->>FVaws: invoke_lambda("subscribe_to_event", full_payload)
alt invoke succeeds
    FVaws-->>Subscribe: response
    Subscribe-->>LambdaHandler: response (returned)
else invoke throws
    Subscribe-->>LambdaHandler: raise NotFoundError("Notification is disabled")
end
```
