    participant Caller
    participant get_storage
    participant Settings as StorageSettings
    participant Provider as StorageProvider
    participant Logger as logger
    participant S3 as S3Storage
    participant Azure as AzureBlobStorage
    participant Dev as DevPassthroughStorage

    Caller->>get_storage: call get_storage(settings)
    get_storage->>Logger: info("Document storage provider resolved: ...")
    alt settings is StorageSettings
        get_storage->>Provider: read settings.provider
        alt provider == AWS
            get_storage->>Logger: info("Document storage provider resolved: AWS")
            Settings-->>get_storage: bucket_name, region
            alt bucket_name missing
                get_storage-->>Caller: raise ValueError("Missing bucket_name...")
            else
                get_storage->>S3: instantiate S3Storage(bucket_name, region or "us-east-1")
                S3-->>get_storage: S3Storage instance
                get_storage-->>Caller: return S3Storage
            end
        else alt provider == AZURE
            get_storage->>Logger: info("Document storage provider resolved: AZURE")
            Settings-->>get_storage: container_name, account_name
            get_storage->>Azure: instantiate AzureBlobStorage(container_name, account_name)
            Azure-->>get_storage: AzureBlobStorage instance
            get_storage-->>Caller: return AzureBlobStorage
        else
            get_storage->>Logger: info("Document storage provider resolved: dev (fallback)")
            get_storage->>Dev: instantiate DevPassthroughStorage()
            Dev-->>get_storage: DevPassthroughStorage instance
            get_storage-->>Caller: return DevPassthroughStorage
        end
    else
        get_storage->>Logger: info("Document storage provider resolved: dev (default)")
        get_storage->>Dev: instantiate DevPassthroughStorage()

        get_storage-->>Caller: return DevPassthroughStorage
    end
